<summary>
- Feature-branch prompts (those with a `branch` field) never trigger a version bump or git tag — they only commit changes and add changelog entries to `## Unreleased`
- After each prompt on a feature branch completes, the processor checks whether any more queued prompts share the same branch
- When the last prompt on a feature branch completes and `pr` is false, the processor merges the feature branch into the default branch and runs a full release (rename `## Unreleased` to version, tag, push)
- When more prompts remain on the same branch, the processor skips the merge and release — the next prompt will continue on the same branch
- Merge conflicts cause a clear error with the feature branch preserved — no silent data loss
- `prompt.Manager` gains a `HasQueuedPromptsOnBranch(ctx, branch, excludePath)` query used by the processor to detect the "last prompt" condition
</summary>

<objective>
Guard releases so they only happen on the default branch: feature-branch prompts commit without releasing, and the full release fires only after the last prompt on the branch merges it to the default branch. This is the release-control layer for spec 029; it builds on the branch-switching infrastructure added in the previous prompt.
</objective>

2. Implement on the concrete `manager` struct:
   ```go
   func (m *manager) HasQueuedPromptsOnBranch(ctx context.Context, branch string, excludePath string) (bool, error) {
       queued, err := m.ListQueued(ctx)
       if err != nil {
           return false, errors.Wrap(ctx, err, "list queued prompts")
       }
       for _, p := range queued {
           if p.Path == excludePath {
               continue
           }
           if p.Branch() == branch {
               return true, nil
           }
       }
       return false, nil
   }
   ```
   Note: at the point this is called, the current prompt has already been moved to `completed/`, so it will not appear in `ListQueued`. The `excludePath` parameter handles the edge case where the file has not yet been moved (call with the current path to be safe).

   func (b *brancher) MergeToDefault(ctx context.Context, branch string) error {
       defaultBranch, err := b.DefaultBranch(ctx)
       if err != nil {
           return errors.Wrap(ctx, err, "get default branch")
       }
       // Ensure we're on the default branch
       // #nosec G204 -- defaultBranch comes from gh CLI, branch comes from validated frontmatter
       if err := exec.CommandContext(ctx, "git", "checkout", defaultBranch).Run(); err != nil {
           return errors.Wrap(ctx, err, "switch to default branch before merge")
       }
       // Merge feature branch into default (no fast-forward to preserve branch history)
       // #nosec G204 -- branch name comes from validated frontmatter
       cmd := exec.CommandContext(ctx, "git", "merge", "--no-ff", branch)
       var stderr strings.Builder
       cmd.Stderr = &stderr
       if err := cmd.Run(); err != nil {
           return errors.Wrapf(ctx, err, "merge branch %q to default: %s", branch, stderr.String())
       }
       slog.Info("merged feature branch to default", "branch", branch, "default", defaultBranch)
       return nil
   }
   ```

       gitCtx context.Context,
       ctx context.Context,
       title string,
       featureBranch string,
   ) error {
       // On a feature branch: commit only, never release
       if featureBranch != "" {
           if err := p.releaser.CommitOnly(gitCtx, title); err != nil {
               return errors.Wrap(ctx, err, "commit on feature branch")
           }
           slog.Info("committed changes on feature branch (no release)", "branch", featureBranch)
           return nil
       }

       // Default branch: original release logic unchanged
       if !p.releaser.HasChangelog(gitCtx) {
           if err := p.releaser.CommitOnly(gitCtx, title); err != nil {
               return errors.Wrap(ctx, err, "commit")
           }
           slog.Info("committed changes")
           return nil
       }
       bump := determineBump()
       nextVersion, err := p.releaser.GetNextVersion(gitCtx, bump)
       if err != nil {
           return errors.Wrap(ctx, err, "get next version")
       }
       if err := p.releaser.CommitAndRelease(gitCtx, bump); err != nil {
           return errors.Wrap(ctx, err, "commit and release")
       }
       slog.Info("committed and tagged", "version", nextVersion)
       return nil
   }
   ```

11. Implement `handleBranchCompletion`:
    ```go
    func (p *processor) handleBranchCompletion(
        gitCtx context.Context,
        ctx context.Context,
        pf *prompt.PromptFile,
        promptPath string,
        title string,
        featureBranch string,
    ) error {
        // Check if any other queued prompts share the same branch
        hasMore, err := p.promptManager.HasQueuedPromptsOnBranch(ctx, featureBranch, promptPath)
        if err != nil {
            slog.Warn("failed to check remaining prompts on branch", "branch", featureBranch, "error", err)
            return nil // non-fatal: skip merge, let next run re-check
        }
        if hasMore {
            slog.Info("more prompts queued on branch — skipping merge", "branch", featureBranch)
            return nil
        }

        slog.Info("last prompt on branch — merging to default and releasing", "branch", featureBranch)

        // Merge feature branch to default (we're already on default after restoreDefaultBranch)
        if err := p.brancher.MergeToDefault(gitCtx, featureBranch); err != nil {
            return errors.Wrap(ctx, err, "merge feature branch to default")
        }

        // Release on default branch (pass empty featureBranch so release logic runs)
        if err := p.handleDirectWorkflow(gitCtx, ctx, title, ""); err != nil {
            return errors.Wrap(ctx, err, "release after branch merge")
        }

        return nil
    }
    ```
